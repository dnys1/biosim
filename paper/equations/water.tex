\section{Water Properties} \label{water_properties}
The following equations are used to estimate the properties of water.
\subsection{Saturation vapor pressure ($e_s$)}
\begin{equation}
    e_s = 0.611\exp{\left[\frac{17.3\,T_\text{air}}{T_\text{air}+237.3}\right]}
\end{equation}
where
\begin{conditions*}
    e_S & Saturation vapor pressure (kPa) \\
    T_\text{air} & Air temperature (\degree C)
\end{conditions*}
\subsection{Water density ($\rho_w$)}
\begin{multline}
    \rho_W = 999.5308 + 6.32693\times10^{-2}T_W - 8.523829\times10^{-3}T_W^2 + \\ 6.943248\times10^{-5}T_W^3 - 3.3821216\times10^{-7}T_W^4
\end{multline}
where
\begin{conditions*}
    \rho_W & Water density (kg/m$^3$) \\
    T_W & Water temperature (\degree C)
\end{conditions*}
\subsection{Latent heat of evaporation ($L$)}
\begin{multline}
\label{eq:latent_heat}
    L \approx 2362\times(-6.178\times10^{-15}\,T_W^6 +
        8.001\times10^{-12}\,T_W^5 + \\
        4.930\times10^{-9}\,T_W^4 +
        5.441\times10^{-7}\,T_W^3 + \\
        4.451\times10^{-5}\,T_W^2 -
        0.562\,T_W+1093.2)
\end{multline}
where
\begin{conditions*}
    L & Latent heat of evaporation (J/kg) \\
    T_W & Water temperature (\degree C)
\end{conditions*}
\subsection{Dynamic viscosity ($\mu$)}
\begin{equation}
    \mu \approx 10^{-3}\times \exp{\left[-3.7188+\frac{578.919}{T_W+135.604}\right]}
\end{equation}
where
\begin{conditions*}
\mu & Dynamic viscosity of water (Pa-s) \\
T_W & Water temperature (\degree C)
\end{conditions*}
